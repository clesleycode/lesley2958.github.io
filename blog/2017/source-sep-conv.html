def gen_sequence(start=0, length =80, p=0.95):
    sequence = np.zeros(length)
    sequence[0] = start
    for i in range(1, length):
        random_num = np.random.rand()
        if random_num > p:
            # Switch state
            sequence[i] = 1-sequence[i-1]
        else:
            sequence[i] = sequence[i-1]
    return sequence
